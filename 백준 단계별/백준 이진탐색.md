## 2110번. 공유기 설치

```python

# 이진탐색 함수
def search(lst):
    # 왼쪽 값, 오른쪽 값 ( 거리의 최대 ) , 중간 설정
    left=0
    right=lst[-1]-lst[0]
    mid=(right-left)//2
    
    # 
    while left<=right:
        # 거리에 따라 설치할 수 있는 공유기 개수 카운트
        cnt=1
        val=lst[0]
        for i in lst[1:]:
            if i >= val+mid:
                val=i
                cnt+=1
        # 조건의 공유기 개수보다 설치를 못하면 (거리를 줄여야 한다면)
        if cnt < C:
            right=mid-1
            mid=(left+right)//2
        # 조건의 공유기 개수를 설치하거나 더 많이 하면 ( 최대를 구해야 하므로 반복)
        else:
            left=mid+1
            mid=(left+right)//2
    return mid

N,C =map(int,input().split())
lst=[]
for _ in range(N):
    lst.append(int(input()))
# 이진탐색을 위한 오름차순 정렬
lst.sort()
print(search(lst))
```

# 1920번
def search(lst,n):
    lower=0
    upper=len(lst)-1
    mid=(lower+upper)//2
    while upper>=lower:
        if n==lst[mid]:
            return 1
        elif n> lst[mid]:
            lower=mid+1
        else:
            upper=mid-1
        mid=(lower+upper)//2
    return 0
