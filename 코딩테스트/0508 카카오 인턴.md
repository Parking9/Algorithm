```python
def solution(n, k, cmd):
    # 현재 선택한 행
    current = k
    # 한 칸 아래로 움직일 때 이동할 칸
    downTo = [i for i in range(1, n + 1)]
    downTo[n - 1] = -1
    # 한 칸 위로 움직 일 때 이동할 칸
    upTo = [i for i in range(-1, n - 1)]
    # 삭제 행들 차례대로 스택
    remove_stack = []
    answer = ['O' for _ in range(n)]
    for command in cmd:
        # 행 선택, 아래로 움직일 때
        if command[0] == 'D':
            # 움직일 칸, 이동
            N = int(command.split()[1])
            for i in range(N):
                current = downTo[current]
        # 위로 움직일 때
        elif command[0] == 'U':
            N = int(command.split()[1])
            for i in range(N):
                current = upTo[current]

        # 삭제 command
        elif command == 'C':
            # 삭제, 스택 추가
            answer[current] = 'X'
            # 맨 마지막 칸이 아닐 때
            if downTo[current] != -1:
                upTo[downTo[current]] = upTo[current]
            # 맨 위 칸이 아닐 때
            if upTo[current] != -1:
                downTo[upTo[current]] = downTo[current]
            # 삭제한 행 스택에 추가
            remove_stack.append(current)

            # 삭제하고 현재 선택된 행 바꿔주기
            # 맨 아래칸이면 선택한 행 위의 행으로, 아니면 아래로 이동
            if downTo[current] == -1:
                current = upTo[current]
            else:
                current = downTo[current]

        # 되돌리기
        elif command == 'Z':
            # 가장 최근에 삭제한 행
            removed = remove_stack.pop()
            answer[removed] = 'O'
            if upTo[removed] != -1:
                downTo[upTo[removed]] = removed
            if downTo[removed] != -1:
                upTo[downTo[removed]] = removed

    return ''.join(answer)
```
