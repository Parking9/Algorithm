from datetime import datetime, timedelta


def solution(n, customers):
    v=[0]*(n+1)
    endtime=[0]*(n+1)
    runtime=[0]*(n+1)
    ans=[0]*(n+1)
    for cus in customers:
        cus=cus.split(' ')
        cus_time=datetime.strptime(cus[0]+' '+cus[1],'%m/%d %H:%M:%S')
        v=[ 0 if et ==0 or et <= cus_time else 1 for et in endtime]
        if set(v) != {1}:
            mv=99999
            midx=0
            for idx in [id for id, value in enumerate(v[1:]) if value==0]:
                if runtime[idx] < mv:
                    mv=runtime[idx]
                    midx=idx
            v[midx]=1
            endtime[midx] = cus_time + timedelta(minutes=int(cus[2]))
            ans[midx]+=1
            runtime[midx]+=int(cus[2])
        else:
            i=sorted([(i,j) for i,j in enumerate(endtime[1:])], key=lambda x: (x[1],x[0]))[0]
            v[i]=1
            endtime[i]=cus_time + timedelta(minutes=int(cus[2]))
            ans[i]+=1
            runtime[i]+=cus[2]
    return max(ans)
# print(solution(2, ['02/28 23:59:00 03','03/01 00:00:00 02','03/01 00:05:00 01']))
