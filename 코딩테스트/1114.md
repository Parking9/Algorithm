import json
import requests


def avgRotorSpeed(statusQuery, parentId):
    url = 'https://jsonmock.hackerrank.com/api/iot_devices/search'+'?status={}'.format(statusQuery)
    response = requests.get(url).json()
    pages = response['total_pages']
    
    ans=0
    cnt=0
    while pages >0 : 
        curr_url = url+'?page={}'.format(pages)
        curr_response = requests.get(curr_url).json()
        curr_data = curr_response['data']
        for i in curr_data:
            if i['parent']['id'].notnull():
                ans+=i['operationParams']['rotorSpeed']
                cnt+=1
    if cnt ==0:
        return 0
    else:
        return int(ans/cnt)
