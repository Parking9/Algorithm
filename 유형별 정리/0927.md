def dfs(x,y,N,M,arr,lst):
    for st in list(filter(lambda v: v+x in range(M-1,M+N-1), list(range(M)))):
        for st2 in list(filter(lambda v: v+y in range(M-1,M+N-1), list(range(M)))):
            if arr[st+x][st2+y]==0:
                if lst[st][st2]==1:
                    arr[st+x][st2+y]=1
                else:
                    return 0
            else:
                if lst[st][st2]==1:
                    return 0
    for j in range(M-1,N+M-1):
        for j2 in range(M-1,N+M-1):
            if arr[j][j2] != 1:
                return 0
    return 1



def solution(key, lock):
    M=len(key)
    N=len(lock)
    for lst in [key,rotate90(key,M),rotate180(key,M),rotate270(key,M)]:
        for x in range(0,M+N-1):
            for y in range(0,M+N-1):
                arr=[[0]*(N+2*M-2) for _ in range(N+2*M-2)]
                for i in range(N):
                    for i2 in range(N):
                        arr[i+M-1][i2+M-1]=lock[i][i2]
                if dfs(x,y,N,M,arr,lst) ==1:
                    return True
    return False

def search(arr,cnt):
    for i in list(zip(*arr)):
        if list(i) ==[0]*5:
            cnt+=1
    for j in arr:
        if j == [0]*5:
            cnt+=1
    v=[]
    v2=[]
    for k in range(5):
        v.append(arr[k][4-k])
        v2.append(arr[k][k])
    if v==[0]*5:
        cnt+=1
    if v2==[0]*5:
        cnt+=1
    return cnt

def turn(n,arr):
    for i in range(5):
