# 다익스트라 알고리즘

```python
T= int(input())
INF=99999999999999
for tc in range(1,T+1):
    N, M, X = map(int,input().split())
    arr=[[INF]*(N) for _ in range(N)]


    for _ in range(M):
        x, y, c = map(int,input().split())
        x-=1
        y-=1
        arr[x][y]=c


    for i in range(N):
        arr[i][i]=0
    D = arr[X]
    U = [0]*N


    cnt=0
    while cnt != N:
        cnt+=1
        minV = INF
        w=0
        for j in range(N):
            if D[j]<minV and U[j] ==0:
                w=j
                minV=D[j]
        U[w]=1

        for k in range(N):
            if U[k]==0:
                if 0<arr[w][k]<INF:
                    D[k]=min(D[k],D[w]+arr[w][k])

    print(D)


```
