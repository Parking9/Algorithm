## 1244 스위치 켜기

```python
#1244 스위치 켜고 끄기
def change1(s,lst):
    cnt=1
    while s*cnt<=N:
        if lst[s*cnt-1]==1:
            lst[s*cnt-1]=0
        else:
            lst[s*cnt-1]=1
        cnt+=1
    return lst

def change2(s,lst):
    s=s-1
    cnt=1
    if lst[s]==1:
        lst[s]=0
    else:
        lst[s]=1
    if s==0 or s==N-1:
        return lst
    if lst[s-1] != lst[s+1]:
        return lst
    else:
        while s-cnt>=0 and s+cnt <N:
            if lst[s-cnt] == lst[s+cnt]:
                if lst[s-cnt]==1:
                    lst[s-cnt]=0
                    lst[s+cnt]=0
                else:
                    lst[s+cnt]=1
                    lst[s-cnt]=1
                cnt+=1
            else:
                return lst
        return lst



N=int(input())
lst=list(map(int,input().split()))
M=int(input())
for _ in range(M):
    s, l = map(int,input().split())
    if s==1:
        lst=change1(l,lst)
    else:
        lst=change2(l,lst)
for i in range(N):
    print(lst[i], end=' ')
    if (i+1)%20==0:
        print()


```

## 2563. 색종이

```python
## 2563 색종이
cnt=0
arr=[[0]*100 for _ in range(100)]
for _ in range(int(input())):
    a,b= map(int,input().split())
    for i in range(a-1,a+9):
        for j in range(b-1,b+9):
            if arr[i][j]==0:
                arr[i][j]=1
                cnt+=1
print(cnt)
```



## 2564번. 경비원

cnt=0

    lst.append(list(map(int,input().split())))
a,b=map(int,input().split())
for i in lst:

                else:

    else:
