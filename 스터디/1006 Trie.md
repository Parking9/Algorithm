```python
class Trienode:
    def __init__(self):
        self.word=False
        self.child={}

class Trie:
    def __init__(self):
        self.root=Trienode()
    
    def insert(self,word : str) -> None:
        node=self.root
        for ch in word:
            if ch not in node.child:
                node.child[ch]=Trienode()
            node=node.child[ch]
        node.word=True

    def search(self, word :str) -> bool :
        node=self.root
        for ch in word:
            if ch in node.child:
                node=node.child[ch]
            else:
                return False
        if node.word==True:
            return True

    def startwith(self, word : str ) -> bool :
        node=self.root
        for ch in word:
            if ch in node.child:
                node=node.child[ch]
            else:
                return False
        return True
```

```python
class TrieNode:
    def __init__(self):
        self.word=False
        self.child={}

class Trie:
    def __init__(self):
        self.root=TrieNode()

    def insert(self,word : str) -> bool:
        node=self.root
        for ch in word:
            if ch not in node.child:
                node.child[ch]=TrieNode()
            node=node.child[ch]
        node.word=True
    def printStructure(self, lv, node):
        if lv==0:
            node=self.root
        for cd in sorted(node.child.keys()):
            print('--'*lv,cd,sep='')
            self.printStructure(lv+1,node.child[cd])

trie=Trie()
N=int(input())
for _ in range(N):
    lst=list(input().split())
    trie.insert(lst[1:])
```
