## 프로그래머스 _ 정삼각형

```python
def solution(triangle):
    for tri in range(1,len(triangle)):
        for i in range(len(triangle[tri])):
            if i in [0,len(triangle[tri])-1]:
                if i==0:
                    triangle[tri][i] +=triangle[tri - 1][i]
                else:
                    triangle[tri][i] +=triangle[tri - 1][i - 1]
            else:
                triangle[tri][i]+=max([triangle[tri-1][i-1],triangle[tri-1][i]])
    return max(triangle[-1])
print(solution([[7], [3, 8], [8, 1, 0], [2, 7, 4, 4], [4, 5, 2, 6, 5]]))
```
